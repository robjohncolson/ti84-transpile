```javascript
import { createCPU } from './cpu-runtime.js';
import { PRELIFTED_BLOCKS } from './ROM.transpiled.js';

const { cpu, mem } = createCPU();
const peripherals = (await import('./peripherals.js')).createPeripheralBus({ timerInterrupt: false });

// Load transpiled blocks
for (const [key, fn] of Object.entries(PRELIFTED_BLOCKS)) {
  cpu.loadBlock(key, fn);
}

// Set up OP1=12.0 and OP2=8.0 in TI FP format
// TI FP: [type, exponent, mantissa...] where exp=0x80+digits_before_decimal
// 12.0 = [0x00, 0x81, 0x12, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00]
// 8.0  = [0x00, 0x80, 0x80, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00]
const OP1 = 0xD005F8;
const OP2 = 0xD00603;
mem[OP1] = 0x00; mem[OP1+1] = 0x81; mem[OP1+2] = 0x12;
for (let i = 3; i < 9; i++) mem[OP1+i] = 0x00;
mem[OP2] = 0x00; mem[OP2+1] = 0x80; mem[OP2+2] = 0x80;
for (let i = 3; i < 9; i++) mem[OP2+i] = 0x00;

// Standard CPU setup
cpu.SP = 0xD1A881;
cpu.IX = 0xD1A860;
cpu.IY = 0xD00080;
cpu.IM = 2;
cpu.ADL = 1;

// Error number location
const errNo = 0xD008DF;
mem[errNo] = 0x00;

// FPS pointer
const fpsPtr = 0xD0258D;
mem[fpsPtr] = 0x00; mem[fpsPtr+1] = 0xAA; mem[fpsPtr+2] = 0xD1;

// Entry point for gcd handler
cpu.PC = 0x068D3D;
```
