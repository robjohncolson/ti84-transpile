## Context

## eZ80 instruction reference
- JP nn = 0xC3 nn24 (4 bytes ADL)
- JP Z,nn = 0xCA nn24 (4 bytes ADL)
- JP NZ,nn = 0xC2 nn24 (4 bytes ADL)
- JR d = 0x18 d (2 bytes, d is signed offset from next instruction)
- JR Z,d = 0x28 d (2 bytes)
- JR NZ,d = 0x20 d (2 bytes)
- CALL nn = 0xCD nn24 (4 bytes ADL)
- RET = 0xC9 (1 byte)
- RET Z = 0xC8 (1 byte)
- RET NZ = 0xC0 (1 byte)
- LD A,(nn) = 0x3A nn24 (4 bytes ADL)
- OR A = 0xB7 (1 byte)
- CP n = 0xFE n (2 bytes)
- BIT b,(IY+d) = 0xFD 0xCB d (0x46+8*b) (4 bytes)
