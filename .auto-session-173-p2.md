# Task: Fix ADL-mode flag bugs in addWithCarryWord and subtractWithBorrowWord

## Context
File: `TI-84_Plus_CE/cpu-runtime.js`
The TI-84 Plus CE uses an eZ80 CPU with 24-bit registers. In ADL mode (`this.madl = 1`), word operations (ADC HL, SBC HL) operate on 24-bit values. The flags should be computed based on the full 24-bit result width when in ADL mode, and 16-bit width when in z80 mode.

The CPU object has `this.addressMask` which returns `0xffffff` in ADL mode and `0xffff` in z80 mode.

## The Bug
Lines 343-367 of `cpu-runtime.js`:

```javascript
addWithCarryWord(a, b) {
    const carry = this._getFlag(FLAG_C) ? 1 : 0;
    const result = a + b + carry;
    const r16 = result & 0xffff;                          // BUG: always 16-bit
    this._setFlag(FLAG_S, r16 & 0x8000);                  // BUG: checks bit 15, not bit 23 in ADL
    this._setFlag(FLAG_Z, r16 === 0);                     // BUG: checks 16-bit zero, not 24-bit
    this._setFlag(FLAG_H, ((a ^ b ^ result) & 0x1000) !== 0);
    this._setFlag(FLAG_PV, ((a ^ result) & (b ^ result) & 0x8000) !== 0);  // BUG: bit 15
    this._setFlag(FLAG_N, false);
    this._setFlag(FLAG_C, result > 0xffff);               // BUG: 16-bit overflow
    return result & this.addressMask;
}

subtractWithBorrowWord(a, b) {
    const carry = this._getFlag(FLAG_C) ? 1 : 0;
    const result = a - b - carry;
    const r16 = result & 0xffff;                          // BUG: always 16-bit
    this._setFlag(FLAG_S, r16 & 0x8000);                  // BUG
    this._setFlag(FLAG_Z, r16 === 0);                     // BUG
    this._setFlag(FLAG_H, ((a ^ b ^ result) & 0x1000) !== 0);
    this._setFlag(FLAG_PV, ((a ^ b) & (a ^ result) & 0x8000) !== 0);  // BUG: bit 15
    this._setFlag(FLAG_N, true);
    this._setFlag(FLAG_C, result < 0);                    // This one is actually OK (sign-based)
    return result & this.addressMask;
}
```

All flag computations use hardcoded 16-bit constants. In ADL mode they should use 24-bit constants.

## The Fix
Use `this.addressMask` and derived bit positions to make flags mode-aware:

For `addWithCarryWord`:
```javascript
addWithCarryWord(a, b) {
    const carry = this._getFlag(FLAG_C) ? 1 : 0;
    const result = a + b + carry;
    const mask = this.addressMask;
    const msb = this.madl ? 0x800000 : 0x8000;
    const halfBit = this.madl ? 0x100000 : 0x1000;
    const masked = result & mask;
    this._setFlag(FLAG_S, masked & msb);
    this._setFlag(FLAG_Z, masked === 0);
    this._setFlag(FLAG_H, ((a ^ b ^ result) & halfBit) !== 0);
    this._setFlag(FLAG_PV, ((a ^ result) & (b ^ result) & msb) !== 0);
    this._setFlag(FLAG_N, false);
    this._setFlag(FLAG_C, result > mask);
    return masked;
}
```

For `subtractWithBorrowWord`:
```javascript
subtractWithBorrowWord(a, b) {
    const carry = this._getFlag(FLAG_C) ? 1 : 0;
    const result = a - b - carry;
    const mask = this.addressMask;
    const msb = this.madl ? 0x800000 : 0x8000;
    const halfBit = this.madl ? 0x100000 : 0x1000;
    const masked = result & mask;
    this._setFlag(FLAG_S, masked & msb);
    this._setFlag(FLAG_Z, masked === 0);
    this._setFlag(FLAG_H, ((a ^ b ^ result) & halfBit) !== 0);
    this._setFlag(FLAG_PV, ((a ^ b) & (a ^ result) & msb) !== 0);
    this._setFlag(FLAG_N, true);
    this._setFlag(FLAG_C, result < 0);
    return masked;
}
```
